\newcommand{\xscale}{0.2}
\newcommand{\yscale}{0.6}
\newcommand{\object}[3]{
	\draw [fill=gray!50] (#1*\xscale,0) rectangle (#2*\xscale+\xscale,\yscale);
	\draw (#1*0.5*\xscale + #2*0.5*\xscale + 0.5*\xscale,\yscale) node[above]{\Var{#3}};
}

\begin{document}
	\begin{tikzpicture}	
		\object{1}{4}{a}
		\object{6}{9}{b}
		\object{17}{23}{c}
		\object{25}{29}{d}
		\object{33}{36}{e}
		\object{39}{45}{f}
		\object{48}{49}{g}
		
		\foreach \x in {1,2,...,49} {
			\draw[gray!30,very thin] (\x*\xscale,0*\yscale) -- (\x*\xscale,1*\yscale);
		}
		\draw [thick] (0,0) rectangle (50*\xscale,1*\yscale);
		
		\foreach \x in {0,10,20,30,40,50} {
			\draw [very thick] (\x*\xscale,1*\yscale) -- (\x*\xscale,0) node[below,font=\footnotesize]{\x};
		}
	\end{tikzpicture}
\end{document}
